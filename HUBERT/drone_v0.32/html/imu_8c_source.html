219 static inline void imu_usart_init( void )
220 {
223 GPIO_InitTypeDef GPIO_InitStructure;
224 USART_InitTypeDef USART_InitStructure;
225 
226  /* Enable GPIO clock */
227  RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOD, ENABLE);
228 
229  /* Enable UART clock */
230  RCC_APB1PeriphClockCmd(RCC_APB1Periph_USART3, ENABLE);
231 
232  /* Connect PXx to USARTx_Rx*/
233 
234  GPIO_PinAFConfig(GPIOD, GPIO_PinSource8, GPIO_AF_USART3);
235  GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;
236  GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_UP;
237  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AF;
238  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
239  GPIO_Init(GPIOD, &GPIO_InitStructure);
240 
241  /* Configure USART Rx as alternate function */
242  GPIO_PinAFConfig(GPIOD, GPIO_PinSource9, GPIO_AF_USART3);
243  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AF;
244  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
245 
246  GPIO_InitStructure.GPIO_Pin = GPIO_Pin_9;
247  GPIO_Init(GPIOD, &GPIO_InitStructure);
248 
249  /* USARTx configured as follow :
250  - BaudRate = 115200 baud
251  - Word Length = 8 Bits
252  - One Stop Bit
253  - No parity
254  - Hardware flow control disabled (RTS and CTS signals)
255  - Receive enabled */
256  USART_InitStructure.USART_BaudRate = 115200;
257  USART_InitStructure.USART_WordLength = USART_WordLength_8b;
258  USART_InitStructure.USART_StopBits = USART_StopBits_1;
259  USART_InitStructure.USART_Parity = USART_Parity_No;
260  USART_InitStructure.USART_HardwareFlowControl = USART_HardwareFlowControl_None;
261  USART_InitStructure.USART_Mode = USART_Mode_Rx;
262 
263  /* USART configuration */
264  USART_Init(USART3, &USART_InitStructure);
265 
266  /* Enable USART */
267  USART_Cmd(USART3, ENABLE);
268 }

271 static inline void imu_receive_frame( uint16_t pusBuff[ FRAME_LEN ] )
272 {
273 uint8_t ucIndex = 0;
274 
278  /* Reception of the IMU string, while the last character of frame has not
279  been received (character of start of next frame) */
280  while( pusBuff[ ucIndex ] != 'S' )
281  {
282  /* Wait for a received byte */
283  while( USART_GetFlagStatus( USART3, USART_FLAG_RXNE ) == RESET )
284  {
285  /* Do nothing */
286  }
287 
288  /* Return to the start of pusBuff if start of frame character is
289  received */
291  if( ucIndex > 0 && USART_ReceiveData( USART3 ) == 'R' )
292  {
293  ucIndex = 0;
294  pusBuff[ ucIndex ] = USART_ReceiveData( USART3 );
295  }
296 
297  pusBuff[ ucIndex ] = USART_ReceiveData( USART3 );
298  ucIndex++;
299  }
300 }

304 static inline void imu_parse_frame( uint16_t pusBuff[ FRAME_LEN ], struct IMUData * const pxIMUData )
305 {
306 uint8_t ucIndex = 0;
307 
308  /* Roll */
309  pxIMUData->plAngle[ IMU_AXIS_X ] = 0;
310  /* Pitch */
311  pxIMUData->plAngle[ IMU_AXIS_Y ] = 0;
312  /* Yaw */
313  pxIMUData->plAngle[ IMU_AXIS_Z ] = 0;
314  /* Altitude */
315  pxIMUData->lAltitude = 0;
316 
317  /* Parsing of the received string to extract numerical values */
318  ucIndex = 0;
319  if( pusBuff[ ucIndex ] == 'R' )
320  {
321  ucIndex++;
322  /* Roll string parsing and update */
323  if( pusBuff[ ucIndex ] == '-' )
324  {
325  ucIndex++;
326  while( pusBuff[ ucIndex ] != 'P' )
327  {
328  pxIMUData->plAngle[ IMU_AXIS_X ] = pxIMUData->plAngle[ IMU_AXIS_X ] * 10 + ( ( int16_t ) pusBuff[ ucIndex ] ) - '0';
329  ucIndex++;
330  }
331  pxIMUData->plAngle[ IMU_AXIS_X ] = -pxIMUData->plAngle[ IMU_AXIS_X ];

338  ucIndex++;
339  }
340  }
341  ucIndex++;
342
